// Countdown Timer
function startCountdown(elementId, hours = 24) {
const endTime = new Date().getTime() + (hours * 60 * 60 * 1000);
setInterval(() => {
const distance = endTime - new Date().getTime();
if (distance < 0) return;
const h = Math.floor(distance / (1000 * 60 * 60 * 24));
const m = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
const s = Math.floor((distance % (1000 * 60)) / 1000);
const el = document.getElementById(elementId);
if (el) el.innerHTML = `⏰ ${h}h ${m}m ${s}s`;
}, 1000);
}

// 5. WISHLIST SYSTEM
let wishlist = JSON.parse(localStorage.getItem('wishlist')) || [];

function toggleWishlist(productName, button) {
if (wishlist.includes(productName)) {
wishlist = wishlist.filter(item => item !== productName);
button.classList.remove('active');
showToast('❌ Removed from wishlist', 'info', 1500);
} else {
wishlist.push(productName);
button.classList.add('active');
showToast('❤️ Added to wishlist!', 'success', 1500);
}
localStorage.setItem('wishlist', JSON.stringify(wishlist));
}

function viewWishlist() {
const modal = document.getElementById('wishlist-modal');
modal.style.display = 'flex';

const wishlistContent = document.getElementById('wishlist-content');
if (wishlist.length === 0) {
wishlistContent.innerHTML = '<p style="text-align: center; color: var(--muted); padding: 40px;">No items in wishlist yet. Add your favorites! ❤️</p>';
} else {
wishlistContent.innerHTML = wishlist.map(item => `<div style="padding: 12px; background: rgba(255,255,255,0.05); border-radius: 8px; margin-bottom: 10px; display: flex; justify-content: space-between; align-items: center;"><span>${item}</span><button onclick="removeFromWishlist('${item}')" style="background: rgba(230,57,70,0.2); border: none; color: var(--primary); padding: 6px 12px; border-radius: 6px; cursor: pointer;">Remove</button></div>`).join('');
}
}

function removeFromWishlist(productName) {
wishlist = wishlist.filter(item => item !== productName);
localStorage.setItem('wishlist', JSON.stringify(wishlist));
viewWishlist();
showToast('❌ Removed from wishlist', 'info', 1500);
}

function clearWishlist() {
if (confirm('Clear all items from wishlist?')) {
wishlist = [];
localStorage.setItem('wishlist', JSON.stringify(wishlist));
viewWishlist();
showToast('🗑️ Wishlist cleared', 'info', 1500);
}
}
